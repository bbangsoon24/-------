<!-- Swiper 라이브러리 연결 (CDN) -->

<script src="https://cdn.jsdelivr.net/npm/swiper/swiper-bundle.min.js"></script>

<script>
  const swiper = new Swiper('.swiper-f', {
    slidesPerView: 3,    // 한 화면에 3개 보임
    slidesPerGroup: 3,   // 버튼 누를 때 3개씩 이동

    loop: false,         // 무한 반복 끔

      nextEl: '.swiper-btn-right',
      prevEl: '.swiper-btn-left',
    },
  });
